import java.applet.*;
import java.awt.*;
  
public class Smiley extends Applet {
    public void paint(Graphics g)
    {
  
        // face outline
        g.drawOval(80, 70, 150, 150);
  
        // eyes
        g.setColor(Color.BLACK);
        g.fillOval(120, 120, 15, 15);
        g.fillOval(170, 120, 15, 15);
  
        // smile
        g.drawArc(130, 180, 50, 20, 180, 180);
    }
}
